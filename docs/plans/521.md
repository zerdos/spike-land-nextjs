# Implementation Plan: Content-to-Ads Loop (Issue #521)

**Epic**: Content-to-Ads Loop - Organic to Paid Pipeline
**Issue**: #521

## Overview

This epic implements a content-to-ads pipeline that identifies high-performing organic social media posts and converts them into paid advertisements. The system will analyze engagement metrics, suggest boost opportunities, automate ad creative generation, and measure incremental impact.

## Stories Breakdown

### ORB-062: Build Boost Detector

Identify top 10% performing organic posts based on engagement metrics.

### ORB-063: Implement Organic-to-Ad Derivation

Convert organic posts to ad creatives with one-click boost functionality and targeting suggestions.

### ORB-064: Create Feedback Loop Analytics

Measure incremental impact of boosted posts vs. organic performance.

```prisma
model BoostCandidate {
  id               String   @id @default(cuid())
  workspaceId      String
  workspace        Workspace @relation(fields: [workspaceId], references: [id], onDelete: Cascade)

  // Link to original organic post
  socialPostId     String
  socialPost       SocialPost @relation(fields: [socialPostId], references: [id], onDelete: Cascade)

  // Performance metrics
  performanceScore Float    // 0-100 score
  percentileRank   Float    // Percentile vs. other posts (0-100)
  engagementRate   Float    // (likes + comments + shares) / impressions

  // Engagement breakdown
  impressions      Int      @default(0)
  likes            Int      @default(0)
  comments         Int      @default(0)
  shares           Int      @default(0)
  clicks           Int      @default(0)

  // Boost status
  status           BoostCandidateStatus @default(PENDING)
  detectedAt       DateTime @default(now())
  reviewedAt       DateTime?
  boostedAt        DateTime?

  // Relations
  boostCampaigns   BoostCampaign[]

  @@index([workspaceId, status])
  @@index([performanceScore])
  @@index([detectedAt])
  @@map("boost_candidates")
}

enum BoostCandidateStatus {
  PENDING      // Detected, awaiting review
  APPROVED     // Approved for boost
  BOOSTED      // Currently running as ad
  REJECTED     // Rejected for boost
  COMPLETED    // Boost campaign completed
}

// Track boosted campaigns derived from organic content
model BoostCampaign {
  id                    String   @id @default(cuid())
  workspaceId           String
  workspace             Workspace @relation(fields: [workspaceId], references: [id], onDelete: Cascade)

  // Link to boost candidate
  boostCandidateId      String
  boostCandidate        BoostCandidate @relation(fields: [boostCandidateId], references: [id], onDelete: Cascade)

  // Ad platform campaign
  platform              AllocatorPlatform
  platformCampaignId    String   // External campaign ID
  campaignName          String

  // Ad creative (generated from organic post)
  adCreative            Json     // Platform-specific ad format

  // Targeting (AI-suggested)
  targeting             Json     // Platform-specific targeting

  // Budget
  budgetCents          Int      // Total budget in cents
  dailyBudgetCents     Int?     // Daily cap

  // Status
  status               BoostCampaignStatus @default(DRAFT)
  startedAt            DateTime?
  endedAt              DateTime?

  // Performance (synced from platform)
  spend                Int      @default(0) // Cents
  impressions          Int      @default(0)
  clicks               Int      @default(0)
  conversions          Int      @default(0)

  createdAt            DateTime @default(now())
  updatedAt            DateTime @updatedAt

  @@index([workspaceId, status])
  @@index([boostCandidateId])
  @@index([platform, platformCampaignId])
  @@map("boost_campaigns")
}

enum BoostCampaignStatus {
  DRAFT        // Created but not launched
  PENDING      // Pending platform approval
  ACTIVE       // Running
  PAUSED       // Temporarily paused
  COMPLETED    // Finished
  FAILED       // Failed to launch
}

// Measure incremental impact of boosting
model BoostAttribution {
  id                    String   @id @default(cuid())

  // Baseline (organic performance before boost)
  baselineImpressions   Int
  baselineEngagement    Int
  baselineClicks        Int
  baselineConversions   Int
  baselinePeriodStart   DateTime
  baselinePeriodEnd     DateTime

  // Boosted performance (during campaign)
  boostedImpressions    Int
  boostedEngagement     Int
  boostedClicks         Int
  boostedConversions    Int
  boostedPeriodStart    DateTime
  boostedPeriodEnd      DateTime

  // Incremental lift calculations
  incrementalImpressions Int     // boosted - baseline projection
  incrementalEngagement  Int
  incrementalClicks      Int
  incrementalConversions Int

  // Efficiency metrics
  costPerIncremental     Float?  // Spend / incremental conversions
  roas                   Float?  // Revenue / spend

  calculatedAt           DateTime @default(now())

  @@index([boostCampaignId])
  @@map("boost_attributions")
}
```

```typescript
// src/lib/boost/scoring.ts
export function calculatePerformanceScore(metrics: {
  impressions: number;
  likes: number;
  comments: number;
  shares: number;
  clicks: number;
}): number {
  // Weighted engagement rate
  const engagementRate = metrics.impressions > 0
    ? ((metrics.likes + metrics.comments * 2 + metrics.shares * 3 + metrics.clicks * 1.5) /
      metrics.impressions) * 100
    : 0;

  // Normalize to 0-100 scale
  // Engagement rates typically 0-10%, so multiply by 10
  return Math.min(100, engagementRate * 10);
}

export function calculatePercentileRank(
  postScore: number,
  allScores: number[],
): number {
  const sorted = [...allScores].sort((a, b) => a - b);
  const rank = sorted.findIndex(s => s >= postScore);
  return (rank / sorted.length) * 100;
}
```

```typescript
export async function suggestTargeting(
```

```typescript
// src/app/api/orbit/[workspaceSlug]/boost/candidates/[id]/boost/route.ts
export async function POST(
  request: Request,
  { params }: { params: { workspaceSlug: string; id: string; }; },
) {
  const { platform, budget, targeting } = await request.json();

  // 1. Load boost candidate and original post
  const candidate = await loadCandidate(params.id);

  // 2. Generate ad creative
  const adCreative = await generateAdCreative(candidate.socialPost, platform);

  // 3. Apply targeting (user-approved or AI-suggested)
  const finalTargeting = targeting || await suggestTargeting(candidate.socialPost);

  // 4. Create platform campaign
  const platformCampaign = await createPlatformCampaign({
    platform,
    adCreative,
    targeting: finalTargeting,
    budget,
  });

  // 5. Create BoostCampaign record
  const boostCampaign = await prisma.boostCampaign.create({
    data: {
      workspaceId: candidate.workspaceId,
      boostCandidateId: candidate.id,
      platform,
      platformCampaignId: platformCampaign.id,
      campaignName: `Boost: ${candidate.socialPost.content.slice(0, 30)}...`,
      adCreative,
      targeting: finalTargeting,
      budgetCents: budget,
      status: "PENDING",
    },
  });

  // 6. Update candidate status
  await prisma.boostCandidate.update({
    where: { id: candidate.id },
    data: { status: "BOOSTED", boostedAt: new Date() },
  });

  return Response.json({ success: true, campaign: boostCampaign });
}
```

**Incremental Lift Calculation:**

```typescript
// src/lib/boost/attribution.ts
export async function calculateIncrementalLift(
  boostCampaign: BoostCampaign,
): Promise<BoostAttribution> {
  // 1. Get baseline organic performance (before boost)
  const baseline = await getBaselineMetrics(
    boostCampaign.boostCandidate.socialPostId,
    boostCampaign.startedAt,
  );

  // 2. Get boosted performance (during campaign)
  const boosted = await getBoostedMetrics(
    boostCampaign.platformCampaignId,
    boostCampaign.platform,
  );

  // 3. Project what organic performance would have been
  const projectedOrganic = projectOrganicPerformance(
    baseline,
    boosted.boostedPeriodEnd.getTime() - boosted.boostedPeriodStart.getTime(),
  );

  // 4. Calculate incremental lift
  const incrementalImpressions = boosted.boostedImpressions - projectedOrganic.impressions;
  const incrementalClicks = boosted.boostedClicks - projectedOrganic.clicks;
  const incrementalConversions = boosted.boostedConversions - projectedOrganic.conversions;

  // 5. Calculate efficiency metrics
  const costPerIncremental = boostCampaign.spend / incrementalConversions;
  const roas = (incrementalConversions * AVERAGE_ORDER_VALUE) / boostCampaign.spend;

  return {
    baselineImpressions: baseline.impressions,
    boostedImpressions: boosted.boostedImpressions,
    incrementalImpressions,
    incrementalClicks,
    incrementalConversions,
    costPerIncremental,
    roas,
  };
}
```

**Metrics Sync Cron Job:**

```typescript
// src/app/api/cron/sync-boost-campaigns/route.ts
export async function GET() {
  // Every hour, sync campaign metrics from platforms
  const activeCampaigns = await prisma.boostCampaign.findMany({
    where: { status: "ACTIVE" },
  });

  for (const campaign of activeCampaigns) {
    const metrics = await fetchPlatformMetrics(
      campaign.platform,
      campaign.platformCampaignId,
    );

    await prisma.boostCampaign.update({
      where: { id: campaign.id },
      data: {
        spend: metrics.spend,
        impressions: metrics.impressions,
        clicks: metrics.clicks,
        conversions: metrics.conversions,
      },
    });

    // Calculate attribution
    const attribution = await calculateIncrementalLift(campaign);
    await prisma.boostAttribution.upsert({
      where: { boostCampaignId: campaign.id },
      create: { boostCampaignId: campaign.id, ...attribution },
      update: attribution,
    });
  }

  return Response.json({ success: true });
}
```
